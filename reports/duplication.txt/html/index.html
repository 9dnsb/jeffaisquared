/**
 * Serialize complex filter objects for logging
 */
function serializeFiltersForLogging(filters: any): Record<string, string | number | boolean | null> {
  if (!filters) return {}

  const serialized: Record<string, string | number | boolean | null> = {}

  for (const [key, value] of Object.entries(filters)) {
    if (value === null || value === undefined) {
      serialized[key] = null
    } else if (typeof value === 'string' || typeof value === 'number' || typeof value === 'boolean') {
      serialized[key] = value
    } else if (value instanceof Date) {
      serialized[key] = value.toISOString()
    } else if (Array.isArray(value)) {
      serialized[key] = value.join(', ')
    } else if (typeof value === 'object') {
      serialized[key] = JSON.stringify(value)
    } else {
      serialized[key] = String(value)
    }
  }

  return serialized
}
